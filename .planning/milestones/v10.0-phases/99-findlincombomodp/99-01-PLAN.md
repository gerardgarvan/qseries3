```cpp
// Solve M·x = b over F_p. M is rows×cols, b has length rows.
// Returns solution x (length cols) if consistent; std::nullopt if inconsistent.
inline std::optional<std::vector<int64_t>> solve_modp(
    std::vector<std::vector<int64_t>> M, const std::vector<int64_t>& b, int64_t p) {
    if (M.empty() || M[0].empty() || static_cast<int>(b.size()) != static_cast<int>(M.size()))
        return std::nullopt;
    int rows = static_cast<int>(M.size());
    int cols = static_cast<int>(M[0].size());
    std::vector<std::vector<int64_t>> A(rows);
    for (int i = 0; i < rows; ++i) {
        A[i].reserve(cols + 1);
        for (int j = 0; j < cols; ++j) A[i].push_back(((M[i][j] % p) + p) % p);
        A[i].push_back(((b[i] % p) + p) % p);
    }
    auto pivot_cols = gauss_to_rref_modp(A, p, cols);  // only pivot on M cols
    for (int i = 0; i < rows; ++i) {
        bool all_zero = true;
        for (int j = 0; j < cols; ++j) {
            if (A[i][j] != 0) { all_zero = false; break; }
        }
        if (all_zero && A[i][cols] != 0) return std::nullopt;
    }
    std::vector<int64_t> x(cols, 0);
    std::vector<int> free_cols;
    for (int j = 0; j < cols; ++j) {
        if (std::find(pivot_cols.begin(), pivot_cols.end(), j) == pivot_cols.end())
            free_cols.push_back(j);
    }
    int pi = 0;
    for (int pc : pivot_cols) {
        if (pi >= rows) break;
        int64_t sum = 0;
        for (int fc : free_cols) sum = (sum + A[pi][fc] * x[fc]) % p;
        x[pc] = ((A[pi][cols] - sum) % p + p) % p;
        ++pi;
    }
    return x;
}
```

Add `findlincombomodp` after `findhommodp`, before `findnonhom`. Pattern: same as `findhomcombo` with degree 1, but uses modp on f and L, and solve_modp instead of solve.

**Algorithm:**
1. Degree 1 monomials = L_1, L_2, ..., L_k (the series in L).
2. T = min(f.trunc, min(L[].trunc)); numCols = T + topshift.
3. f_mod = f.truncTo(T).modp(p); L_mod = each L_i.truncTo(T).modp(p).
4. M = build_matrix(L_mod, numCols) — rows = L_i, cols = q-powers.
5. MT = transpose(M).
6. f_coeffs = f_mod.coeffList(0, numCols-1), convert to int64_t mod p.
7. Return solve_modp(MT, f_coeffs, p).

**Result type:** `std::optional&lt;std::vector&lt;int64_t&gt;&gt;` — coeffs c_1..c_k or nullopt.

**Implementation sketch:**
- Use `build_matrix` on L_mod (after modp). Convert Frac matrix to int64_t mod p (same pattern as findhommodp: val from Frac, then `((val % p) + p) % p`).
- Transpose to get MT (rows = numCols, cols = k).
- f_coeffs: `f_mod.coeffList(0, numCols-1)`, resize to numCols, convert each Frac to int64_t mod p (Frac must have den==1 for modp output).
- Call `solve_modp(MT, f_coeffs_int64, p)`.
- Handle topshift: use `topshift = 0` for initial impl; match findhomcombo by taking T from min truncs and numCols = T + topshift. Accept `(f, L, p, topshift)` to mirror findhommodp, OR `(f, L, p, T)` per user spec. User spec: findlincombomodp(f, L, p, T) with T as 4th arg. So:
  - 4-arg: f, L, p, T — truncate f and L to T, topshift=0, numCols=T.
  - 3-arg: f, L, p — T from min(f.trunc, L[].trunc), topshift=0, numCols=T.

Add function:
```cpp
inline std::optional&lt;std::vector&lt;int64_t&gt;&gt; findlincombomodp(
    const Series&amp; f, const std::vector&lt;Series&gt;&amp; L, int p, int topshift = 0) {
    if (L.empty()) return std::nullopt;
    int T = f.trunc;
    for (const auto&amp; s : L) T = std::min(T, s.trunc);
    int numCols = T + topshift;
    if (numCols <= 0) return std::nullopt;
    auto f_mod = f.truncTo(T).modp(p);
    std::vector&lt;Series&gt; L_mod;
    for (const auto&amp; s : L) L_mod.push_back(s.truncTo(T).modp(p));
    auto M_frac = build_matrix(L_mod, numCols);
    int rows_m = static_cast&lt;int&gt;(M_frac.size());
    int cols_m = rows_m > 0 ? static_cast&lt;int&gt;(M_frac[0].size()) : 0;
    std::vector&lt;std::vector&lt;int64_t&gt;&gt; MT(cols_m, std::vector&lt;int64_t&gt;(rows_m));
    for (int i = 0; i &lt; rows_m; ++i)
        for (int j = 0; j &lt; cols_m; ++j) {
            int64_t val = M_frac[i][j].num.d.empty() ? 0 : static_cast&lt;int64_t&gt;(M_frac[i][j].num.d[0]);
            if (M_frac[i][j].num.neg) val = -val;
            MT[j][i] = ((val % p) + p) % p;
        }
    std::vector&lt;Frac&gt; fc = f_mod.coeffList(0, numCols - 1);
    if (static_cast&lt;int&gt;(fc.size()) &lt; numCols) fc.resize(numCols, Frac(0));
    std::vector&lt;int64_t&gt; b(cols_m);
    for (size_t j = 0; j &lt; static_cast&lt;size_t&gt;(cols_m); ++j) {
        int64_t val = fc[j].num.d.empty() ? 0 : static_cast&lt;int64_t&gt;(fc[j].num.d[0]);
        if (fc[j].num.neg) val = -val;
        b[j] = ((val % p) + p) % p;
    }
    return solve_modp(MT, b, static_cast&lt;int64_t&gt;(p));
}
```

Note: MT has rows = numCols (q-powers), cols = k (L_i). b has length numCols. So we solve MT * c = b, c has length k. Correct.

inline std::string formatFindlincombomodp(const FindlincombomodpResult&amp; r) {
    if (!r.coeffs) return "no linear combination mod " + std::to_string(r.p);
    std::string out;
    for (size_t i = 0; i < r.coeffs->size(); ++i) {
        if (i) out += " + ";
        out += std::to_string((*r.coeffs)[i]) + " L" + std::to_string(i + 1);
    }
    return out + " (mod " + std::to_string(r.p) + ")";
}
